CREATE PROCEDURE [dbo].[VerifyUserLogin]
    @Username NVARCHAR(50), 
    @Password NVARCHAR(50),
    @UserType NVARCHAR(10) OUTPUT,
    @ForeignID INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @UserID INT;

    -- Check User WALA table
    SELECT 
        @UserID = UserID,
        @UserType = UserType,
        @ForeignID = ForeignID
    FROM 
        [User]
    WHERE
        Username = @Username
        AND Password = @Password;

    IF @UserID IS NULL
    BEGIN
        SET @UserType = 'Invalid';
        SET @ForeignID = NULL;
        RETURN;
    END

    -- if Admin
    IF @UserType = 'Admin'
    BEGIN
        IF EXISTS (
            SELECT 1 
            FROM Admin 
            WHERE AdminID = @ForeignID 
        )
        BEGIN
            RETURN;
        END
    END
    -- if Teacher
    ELSE IF @UserType = 'Teacher'
    BEGIN
        IF EXISTS (
            SELECT 1 
            FROM Teacher 
            WHERE TeacherID = @ForeignID 
        )
        BEGIN
            RETURN;
        END
    END
    -- if Student
    ELSE IF @UserType = 'Student'
    BEGIN
        IF EXISTS (
            SELECT 1 
            FROM Student 
            WHERE StudentID = @ForeignID 
        )
        BEGIN
            RETURN; 
        END
    END

    -- If no match found
    SET @UserType = 'Invalid';
    SET @ForeignID = NULL;
END;
